"Start Symbol" = <S>
<S> ::= <STATEMENT>
<STATEMENT> ::= <ASSIGNMENT> | <CONDITIONAL> | <LOOP>
<ASSIGNMENT> ::= <ID> '=' <EXPR> | <S>
<CONDITIONAL> ::= 'if' <EXPR>  '{' <S>  '}' <CONDITIONALP>
<CONDITIONALP> ::= 'else' '{' <S> '}' | <>
<LOOP> ::= 'while' <EXPR> '{' <S> '}' | <S>
<OP> ::= '+' | '-' | '*' | '/' | '=' | '>' | '>=' | ':<=' | ':<' | '!='     
<EXPR> ::= <ID> | <ID> <OP> <ID> | <ID> <OP> <NUMBER> | <NUMBER> <OP> <NUMBER> | <NUMBER> <OP> <ID>
<NUMBER> ::= 1 <NUM> | 2 <NUM> | 3 <NUM> | 4 <NUM>
<NUM> ::= 1 <NUM> | 2 <NUM>| 3 <NUM> | 4 <NUM> | <>
<ID> ::= '_ID'
